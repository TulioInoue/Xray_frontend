3.2) Evaluating for testing set¶

In [7]:
if X_test.max() > 1.0:
    X_test = X_test.astype("float32") / 255.0
if y_test.max() > 1.0:
    y_test = y_test.astype("float32") / 255.0

dice_scores = []
iou_scores = []

for i in range(len(X_test)):
    
    input_data = np.expand_dims(X_test[i], axis=0)
    
    
    raw_result = session.run([output_name], {input_name: input_data})
    predict = raw_result[0][0]
    
    
    predict_binario = (predict > 0.5).astype(np.uint8)
    gabarito = (y_test[i] > 0.5).astype(np.uint8)
    
    
    intersection = np.sum(predict_binario * gabarito)
    total_pixels = np.sum(predict_binario) + np.sum(gabarito)
    union = np.sum(np.maximum(predict_binario, gabarito))
    
    
    smooth = 1e-5
    
    dice = (2.0 * intersection + smooth) / (total_pixels + smooth)
    dice_scores.append(dice)
    
    iou = (intersection + smooth) / (union + smooth)
    iou_scores.append(iou)

print("Metrics:")

print(f"Average Dice Coefficient: {np.mean(dice_scores):.4f}")
print(f"Average IoU (Jaccard):    {np.mean(iou_scores):.4f}")

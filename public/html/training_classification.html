def plot_history(history, title):
  fig, axs = plt.subplots(
    figsize = (20, 5),
    ncols = 2,
  )

  axs[0].plot(
      history.epoch, history.history["val_loss"], "C1", label = "val_loss"
  )
  axs[0].plot(
      history.epoch, history.history["loss"], "C0", label = "loss"
  )

  axs[1].plot(
      history.epoch, history.history["val_sparse_categorical_accuracy"], "C1", label = "val_sparse_categorical_accuracy"
  )
  axs[1].plot(
      history.epoch, history.history["sparse_categorical_accuracy"], "C0", label = "sparse_categorical_accuracy"
  )

  axs[0].set_title(
      label = "loss evolution",
      fontsize = 16,
      fontweight = "bold",
  )

  axs[0].spines[["top", "right"]].set_visible(False)
  axs[0].set_xlabel(xlabel = "epoch")
  axs[0].set_ylabel(ylabel = "loss")
  axs[0].legend()

  axs[1].set_title(
      label = "sparse_categorical_accuracy evolution",
      fontsize = 16,
      fontweight = "bold",
  )

  axs[1].spines[["top", "right"]].set_visible(False)
  axs[1].set_xlabel(xlabel = "epoch")
  axs[1].set_ylabel(ylabel = "sparse_categorical_accuracy")
  axs[1].legend()

  plt.suptitle(t = title, fontsize = 24, fontweight = "bold")
  plt.show()

model = tf.keras.Sequential([
    tf.keras.layers.InputLayer(
        shape = (256, 256, 1)
    ),
    tf.keras.layers.Rescaling(1./255),
    tf.keras.layers.Conv2D(filters = 32, kernel_size=(2, 2), activation = "relu"),
    tf.keras.layers.AveragePooling2D(pool_size = (2, 2)),
    tf.keras.layers.Dropout(0.2),
    tf.keras.layers.Conv2D(filters = 32, kernel_size=(2, 2), activation = "relu"),
    tf.keras.layers.AveragePooling2D(pool_size = (2, 2)),
    tf.keras.layers.Dropout(0.2),
    tf.keras.layers.Conv2D(filters = 32, kernel_size=(2, 2), activation = "relu"),
    tf.keras.layers.AveragePooling2D(pool_size = (2, 2)),
    tf.keras.layers.Dropout(0.2),
    tf.keras.layers.Conv2D(filters = 32, kernel_size=(2, 2), activation = "relu"),
    tf.keras.layers.AveragePooling2D(pool_size = (2, 2)),
    tf.keras.layers.Dropout(0.2),
    tf.keras.layers.Conv2D(filters = 32, kernel_size=(2, 2), activation = "relu"),
    tf.keras.layers.AveragePooling2D(pool_size = (2, 2)),
    tf.keras.layers.Dropout(0.2),
    tf.keras.layers.Flatten(),
    tf.keras.layers.Dense(units = 4, activation = "softmax")
])

model.compile(
    optimizer = tf.keras.optimizers.Adam(1E-4),
    loss = tf.keras.losses.SparseCategoricalCrossentropy(),
    metrics = [tf.keras.metrics.SparseCategoricalAccuracy()],
)

model.summary(show_trainable = True)

import os

    "./drive/MyDrive/chest-xray/checkpoints/first_model.keras",
    "./drive/MyDrive/chest-xray/checkpoints/second_model.keras",
    "./drive/MyDrive/chest-xray/checkpoints/third_model.keras",

for path in paths:
    print(f"Lendo e preparando o modelo: {path}")
    try:
        # 1. Carrega o modelo Keras original
        model = tf.keras.models.load_model(path)

        # 2. Caminhos temporários para o formato SavedModel do TensorFlow
        temp_saved_model_dir = "./temp_saved_model"
        onnx_path = path.replace('.keras', '.onnx')

        # 3. Exporta para SavedModel (o formato de produção nativo do TF)
        # Isso limpa totalmente qualquer amarração ou bug do Keras 3
        model.export(temp_saved_model_dir)

        # 4. Chama o pacote tf2onnx direto pelo terminal do sistema
        # --saved-model: aponta para a pasta que criamos
        # --output: o caminho do arquivo .onnx final
        # --opset 15: garante compatibilidade para rodar no AWS Lambda
        print(f"Convertendo via CLI do tf2onnx...")
        os.system(f"python -m tf2onnx.convert --saved-model {temp_saved_model_dir} --output {onnx_path} --opset 15")

        # Limpa a pasta temporária criada
        os.system(f"rm -rf {temp_saved_model_dir}")

        print(f"  Sucesso absoluto! Arquivo limpo gerado em: {onnx_path}\n")

    except Exception as e:
        print(f"  Erro ao converter o modelo {path}: {e}\n")
